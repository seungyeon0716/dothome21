01. 선언적 함수 

{
    function func(){
        document.write("함수가 실행되었습니다.");
    }
    func();
}

02. 익명 함수 

{
    const func = function(){
        document.write("함수가 실행되었습니다.");
    }
    func();
}

03. 매개변수 함수 

{
    function func(str){
        document.write(str);
    }
    func("함수가 실행되었습니다.");
}

04. 리턴값 함수 

{
    function func(){
        const str = "함수가 실행되었습니다.";
        return str;
    }
    document.write(func());
}

05. 화살표 함수 : 선언적 함수 

{
    func = () => {
        document.write("함수가 실행되었습니다.!")
    }
    func();
}

06. 화살표 함수 : 익명 함수 

{
    const func = () => { 
        document.write("함수가 실행되었습니다")
    }
    func();
}

07. 화살표 함수 : 리턴값 함수 

{
    func= () => {

    document.write(func());
}

12. 내부 함수

{
    function func(){

        funcA();

            document.write("함수B가 실행되었습니다.")

    func();

    function func(strA, strB){

        document.write(arguments[1]);
    }
    func("함수A가 실행되었습니다.", "함수B가 실행되었습니다.");

16. 재귀 함수 

{
    function func(num){
        if(num <= 1){
            document.write("함수가 실행되었습니다.");
        } else {
            document.write("함수가 실행되었습니다.");
            func(num-1);
        }
    }
    func(5);
}

17. 콜백 함수 

{
    function funcA(){
        document.write("함수A가 실행되었습니다.");
    }
    function funcB(str){
        document.write("함수B가 실행되었습니다.");
        str();  //funcA();
    }
    funcB(funcA);
}

18. 콜백 함수 : 반복문 

{
    function func(){
        document.write("함수가 실행되었습니다.");
    }
    function callback(str){
        for(let i=1; i<=5; i++){
            str(i); 
        }
    }
    callback(func);
}

19. 콜백 함수 : 동기/비동기 

{
    //01
    function funcA(){
        console.log("funcA가 실행되었습니다.")
    }
    function funcB(){
        console.log("funcB가 실행되었습니다.")
    }
    funcB();
    funcA();
    
    //02
    function funcC(){
        setTimeout(() => {
            console.log("funcC 실행되었습니다.");
        }, 1000);
    }
    function funcD(){
        console.log("funcD 실행되었습니다.")
    }
    funcC();
    funcD();

    //03
    function funcE(callback){
        setTimeout(() => {
            console.log("funcE 실행되었습니다.");
            callback();
        }, 1000);
    }
    function funcF(){
        console.log("funcF 실행되었습니다.");
    }
    funcE(funcF);
}

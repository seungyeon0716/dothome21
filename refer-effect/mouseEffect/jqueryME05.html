.moveWrap {

    justify-content: center;

    position: relative;
}
.moveImage .mImg {
     width: 50vw; 
     height: 60vh;
     overflow: hidden;
     position: relative;
     transition: transform 500ms cubic-bezier(0.25, 0.46, 0.45, 0.84);
}

     transform: scale(1.025);
}

    width: 100%;
    position: absolute;

    height: 110%;
    object-fit: cover;

}

.cursor {
    position: absolute;
    width: 20px; height: 20px;
    background: #fff;
    border-radius: 50%;
    z-index: 10000;
    user-select: none;
    pointer-events: none;
}

 $(".moveImage").mousemove(function(e){
    gsap.to(".cursor", {duration: .2, left: e.pageX -10, top: e.pageY -10});

    // 1. 좌표값을 가운데로 정렬
    let pageX = e.pageX;
    let pageY = e.pageY;

    // 2. img x축 마우스 값 출력(/20) y축 마우스 값 출력(/20)
    let standardX = $(window).width()/2 - pageX;
    let standardY = $(window).height()/2 - pageY;

    // 이미지
    $(".mImg img").attr({ "style": "transform: translate(" + standardX/20 + "px," + standardY/20 + "px)"});

    // 출력
    $(".pageX").text(pageX);
    $(".pageY").text(pageY);
    $(".standardX").text(standardX);
    $(".standardY").text(standardY);
});

  $(".paraNav ul li").click(function(e){
    e.preventDefault();
    let target = $(this);
    let index = target.index();
    let section = $(".paraCont > section").eq(index);
    let offset = section.offset().top;
    $("html, body").animate({ scrollTop : offset }, 600);
});


$(window).scroll(function(){
    let scrollTop = $(window).scrollTop();

    $(".content-item").each(function(index){
        if( scrollTop >= $(this).offset().top -2){
            $(".paraNav ul li").eq(index).addClass("active").siblings().removeClass("active");
        }
    });

    // info
    $(".paraScroll span").text(parseInt(scrollTop));

    for( let i=1; i<10; i++ ){
        $(".offset"+i).text(parseInt($("#section"+i).offset().top) + "px");
    }
});

$(".paraDot ul li").click(function(e){
    e.preventDefault();
    let target = $(this);       //타켓 설정 : 사용자가 클릭한 버튼을 저장
    let index = target.index()  //인덱스 설정 
    // alert(index) ;
    let section = $(".content-item").eq(index);
    let offset = section.offset().top;
    //alert(offset);
    $("html, body").animate({scrollTop : offset}, 600, "easeInBack");     //scroll 부드럽게 이동
});

$(window).scroll(function(){
    let scrollTop = $(this).scrollTop();

    $(".paraScroll span").text(parseInt(scrollTop));

    $(".content-item").each(function(index){
        if( scrollTop >= $(this).offset().top -2){
            $(".paraDot ul li").eq(index).addClass("active").siblings().removeClass("active");
        }
    });
});
